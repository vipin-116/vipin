let userScore = 0; let computerScore = 0; function play(userChoice) { const choices = ['stone', 'paper', 'scissors']; const computerChoice = choices[Math.floor(Math.random() * 3)]; let result = ""; if (userChoice === computerChoice) { result = "It's a draw!"; } else if ( (userChoice === "stone" && computerChoice === "scissors") || (userChoice === "paper" && computerChoice === "stone") || (userChoice === "scissors" && computerChoice === "paper") ) { result = `You win! ${userChoice} beats ${computerChoice}`; userScore++; } else { result = `You lose! ${computerChoice} beats ${userChoice}`; computerScore++; } document.getElementById("result").innerText = result; document.getElementById("score").innerText = `You: ${userScore} | Computer: ${computerScore}`; }
